Check port:

netstat -aof | findstr 5432

---
# Create Network

docker create network library-network

# Rodar container Postgree na versão 18.3
docker run --name librarydb -e POSTGRES_PASSWORD=postgres -e POSTGRES_USER=postgres -e POSTGRES_DB=library -p 5432:5432 --network library-network -d postgres:18.3

# Rodar Pgadmin4

docker run --name pgadmin4 -e PGADMIN_DEFAULT_EMAIL=admin@admin.com -e PGADMIN_DEFAULT_PASSWORD=admin -p 15432:80 --network library-network -d dbpage/pgadmin4 9.14.0
